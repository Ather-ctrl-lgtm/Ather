 
    Create Tables 
CREATE TABLE Physician (     reg_no INT PRIMARY KEY,     name VARCHAR(30),     tel_no VARCHAR(15),     city VARCHAR(20) 
); 
 
CREATE TABLE Patient (     p_name VARCHAR(30) PRIMARY KEY,     street VARCHAR(30),     city VARCHAR(20) 
); 
 
CREATE TABLE Visit (     p_name VARCHAR(30), 
    reg_no INT,     date_of_visit DATE, 
    fees INT, 
    FOREIGN KEY (p_name) REFERENCES Patient(p_name), 
    FOREIGN KEY (reg_no) REFERENCES Physician(reg_no) 
);

    Get patient details of 'Nashik' city 
SELECT *  
FROM Patient 
WHERE city = 'Nashik'; 
 
    Count number of physicians from 'Mumbai' 
SELECT COUNT(*) AS Mumbai_Physicians 
FROM Physician 
WHERE city = 'Mumbai'; 
 
    Display list of patient names in ascending order 
SELECT p_name  
FROM Patient 
ORDER BY p_name ASC; 
 
    Get patient name and fees, who paid fees > 5000 
SELECT p_name, fees 
FROM Visit 
WHERE fees > 5000; 
 
    Find name and city of patients who visited physician on 13th July 2023 
SELECT p.p_name, p.city 
FROM Patient p 
JOIN Visit v ON p.p_name = v.p_name 
WHERE v.date_of_visit = '2023-07-13'; 
 
    Get name of physician and number of patients visited to him 
SELECT ph.name AS Physician_Name, COUNT(v.p_name) AS No_of_Patients 
FROM Physician ph 
JOIN Visit v ON ph.reg_no = v.reg_no 
GROUP BY ph.name; 
 
    1  Get date-wise total fees collected 
SELECT date_of_visit, SUM(fees) AS Total_Fees 
FROM Visit 
GROUP BY date_of_visit 
ORDER BY date_of_visit;
